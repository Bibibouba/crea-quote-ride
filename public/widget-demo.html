Démo d'Intégration du Widget VTC

Messages du Widget:

Comment intégrer le widget:

<iframe 
  src="https://votre-app.com/widget/DRIVER_ID?departure=Paris&destination=Lyon" 
  width="100%" 
  height="700" 
  frameborder="0" 
  allow="geolocation"></iframe>

<script>
  window.addEventListener('message', function(event) {
    // Vérifier l'origine pour des raisons de sécurité
    if (event.origin !== 'https://votre-app.com') return;
    
    const { event: eventType, data } = event.data;
    
    switch (eventType) {
      case 'QUOTE_READY':
        // Le devis est prêt avec l'ID et le montant
        console.log('Devis prêt:', data.quoteId, data.amount);
        break;
      
      case 'QUOTE_ERROR':
        // Une erreur s'est produite
        console.error('Erreur de devis:', data.message);
        break;
      
      case 'STEP_CHANGED':
        // L'utilisateur a changé d'étape dans le processus
        console.log('Étape changée:', data.step, data.name);
        break;
    }
  });
</script>
